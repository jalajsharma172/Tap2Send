            // Numbers 0-9
            if (e.key >= '0' && e.key <= '9') {
                e.preventDefault();
                addDigit(e.key);
            }
            // Plus key
            else if (e.key === '+' || (e.shiftKey && e.key === '=')) {
                e.preventDefault();
                addDigit('+');

            // Backspace

        function addDigit(digit) {
            if (phoneDisplay.value.length < 15) {
                phoneDisplay.value += digit;
                phoneDisplay.focus(); // Keep focus on input
                // Animate corresponding key button if clicked
                if (event && event.target) {
                    animateKey(event.target);
                }

        }

        function backspace() {
            phoneDisplay.value = phoneDisplay.value.slice(0, -1);
            phoneDisplay.focus(); // Keep focus on input
            if (event && event.target) {
                animateKey(event.target);
            }
        }

            phoneDisplay.value = '';

            }
        }
